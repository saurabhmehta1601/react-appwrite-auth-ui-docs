The FormControl is combination of label with input field. 

```jsx
import { FormControl } from 'react-appwrite-ui';

export default function App() {
  const [email,setEmail] = useState('')
  return (
    <FormControl
      label="Email"
      placeholder="Enter your email"
      type='email'
      value={email}
      onChange={(e) => {setEmail(e.target.value)}}
    />
```
